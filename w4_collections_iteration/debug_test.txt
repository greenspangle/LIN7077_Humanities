def hms_to_s(hours,minutes,seconds):
    seconds_in_hours = hours*60
    seconds_in_minutes = minutes*60
    seconds_in_seconds = seconds
    total_seconds = seconds_in_hours + seconds_in_minutes + seconds_in_seconds
    return total_seconds

def s_to_hms(seconds):
    hours = seconds//3600
    minutes = seconds//60
    seconds = seconds%60
    return (hours, minutes, seconds)
